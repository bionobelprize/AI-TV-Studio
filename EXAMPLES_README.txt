key_concepts = """

┌─ 🔑 关键概念说明 ──────────────────────────────────────────────┐
│                                                                 │
│  [数据结构构建]                                                  │
│  ├─ 创建内存中的对象：Character、Episode、Scene、Shot          │
│  ├─ 不涉及任何API调用                                           │
│  ├─ 消耗：CPU + 内存                                             │
│  ├─ 成本：$0                                                    │
│  ├─ 位置标记：🟢 [数据结构构建]                                  │
│  └─ 示例：                                                      │
│     character = Character(id="char_1", name="林晓")  # 此处      │
│     episode.scenes.append(scene)                    # 此处      │
│                                                                 │
│  [LLM调用]                                                      │
│  ├─ 调用LLM服务生成脚本                                         │
│  ├─ 消耗：LLM token                                              │
│  ├─ 成本：$0.01-$0.30 per episode                               │
│  ├─ 位置标记：🟡 [LLM调用]                                       │
│  ├─ 步骤：脚本生成 (stage 2)                                     │
│  └─ 示例：                                                      │
│     episode = studio.script_generator.generate_episode()  # 此处  │
│                                                                 │
│  [视频模型调用] ⚠️ 高成本                                        │
│  ├─ 调用视频API生成视频                                         │
│  ├─ 消耗：视频API额度/token                                     │
│  ├─ 成本：¥2-5 per shot，¥50-500 per episode                   │
│  ├─ 位置标记：🔴 [视频模型调用] 💰                               │
│  ├─ 步骤：视频生成 (stage 4)                                     │
│  └─ 示例：                                                      │
│     result = studio._mcp_server.call_tool("generate_text_to_video", ...)  │
│                                                                 │
│  [视频组装]                                                      │
│  ├─ FFmpeg处理：合并、转场、混音                                │
│  ├─ 消耗：CPU + 磁盘I/O                                          │
│  ├─ 成本：$0（本地处理）                                        │
│  ├─ 位置标记：🟣 [视频组装]                                      │
│  └─ 示例：                                                      │
│     output_path = studio.video_assembler.assemble_episode(episode)  │
│                                                                 │
